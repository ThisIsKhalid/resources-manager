Control Flow — All Forms

// ── IF — no parentheses!
if x > 0 {
    fmt.Println("positive")
} else if x < 0 {
    fmt.Println("negative")
} else {
    fmt.Println("zero")
}

// Init statement — very idiomatic Go
if err := doWork(); err != nil {   // err scoped to this if block only
    return err
}
if v, ok := cache[key]; ok {       // check existence + use in one line
    return v
}

// ── FOR — Go's ONLY loop (replaces while, do-while, forEach)
for i := 0; i < 10; i++ {}        // classic
for i < 10 { i++ }                // while
for { break }                      // infinite loop
for i, v := range slice {}         // index + value (like forEach)
for k, v := range myMap {}         // map iteration
for _, v := range slice {}         // value only (ignore index)
for ch := range myChan {}          // read channel until closed
for range 5 {}                     // iterate N times (Go 1.22+)

// Break/continue with labels (nested loops)
outer:
for i := 0; i < 5; i++ {
    for j := 0; j < 5; j++ {
        if j == 3 { continue outer }  // skip to next i
        if i == 4 { break outer }     // exit both loops
    }
}

// ── SWITCH — no break needed, no fallthrough by default
switch day {
case "Mon", "Tue", "Wed", "Thu", "Fri":
    fmt.Println("Weekday")
case "Sat", "Sun":
    fmt.Println("Weekend")
default:
    fmt.Println("Unknown")
}

// Expressionless switch (like if-else chain)
switch {
case score >= 90: fmt.Println("A")
case score >= 80: fmt.Println("B")
default:          fmt.Println("F")
}

// Switch with init
switch os := runtime.GOOS; os {
case "linux":   fmt.Println("Linux")
case "darwin":  fmt.Println("macOS")
default:        fmt.Println(os)
}

// ── DEFER — runs when function returns (LIFO order)
func processFile(name string) error {
    f, err := os.Open(name)
    if err != nil { return err }
    defer f.Close()       // guaranteed cleanup — like finally
    defer mu.Unlock()     // runs before f.Close()
    mu.Lock()
    // ... work ...
    return nil
}

Timeline: Week 2

Slices — Internals & All Operations

A slice = 3 fields: pointer (to backing array) + len + cap. Slicing creates a new header pointing to the same backing array. 

// Creating
var s []int               // nil slice (len=0, cap=0, ptr=nil)
s  = []int{}              // empty (not nil)
s  = []int{1, 2, 3}       // literal
s  = make([]int, 5)       // len=5, cap=5, all zeros
s  = make([]int, 0, 100)  // len=0, cap=100 (pre-allocate)

// Appending — ALWAYS reassign! May return new backing array
s = append(s, 4)
s = append(s, 5, 6, 7)
s = append(s, other...)   // spread another slice

// Slicing — SHARES backing array!
a := []int{0, 1, 2, 3, 4}
b := a[1:4]   // [1,2,3] — shares memory with a
b[0] = 99     // modifies a[1]! a is now [0,99,2,3,4]

// Safe independent copy
safe := make([]int, len(a))
copy(safe, a)

// Full slice expression — limits cap of result
limited := a[1:3:3]  // len=2, cap=2 — append won't overwrite a

// Deleting elements
i := 2
s = append(s[:i], s[i+1:]...)          // order preserved, O(n)
s[i] = s[len(s)-1]; s = s[:len(s)-1]   // unordered, O(1)

// Sorting
sort.Ints(nums)
sort.Strings(strs)
sort.Slice(users, func(i, j int) bool {
    return users[i].Name < users[j].Name
})
sort.SliceStable(items, less) // preserves equal-key order

// slices package (Go 1.21+)
slices.Contains(s, 3)
slices.Index(s, 3)         // -1 if not found
slices.Sort(s)
slices.Reverse(s)
slices.Compact(s)          // remove consecutive duplicates
slices.Clone(s)            // shallow copy

Maps

// Creating
var m map[string]int              // nil — READ ok, WRITE panics!
m  = map[string]int{}             // empty, safe to use
m  = make(map[string]int)         // same

// Frequency counter

Structs — Complete

// ── Basic struct
type User struct {
    ID        int        // exported (Uppercase = public)
    Name      string
    Email     string
    age       int        // unexported (lowercase = this package only)
    CreatedAt time.Time
}

// ── Struct tags — used by json, gorm, validate, etc.
type Product struct {
    ID    int     `json:"id"               db:"id"`
    Name  string  `json:"name"             validate:"required,min=3"`
    Price float64 `json:"price"            validate:"required,gt=0"`
    Skip  string  `json:"-"`              // always omit from JSON
    Notes *string `json:"notes,omitempty"` // omit if nil
}

// ── Creating
u1 := User{ID: 1, Name: "Alice"}     // named fields (always prefer this)
u2 := &User{Name: "Bob"}              // pointer to struct
u3 := User{}                          // all zero values

// ── Embedding — Go's composition (replaces inheritance)
type Address struct {
    Street, City, Zip string
}

type Employee struct {
    User               // embedded — User's fields/methods PROMOTED
    Address            // embedded — Address's fields PROMOTED
    Department string
    Salary     float64
}

emp := Employee{User: User{Name: "Alice"}, Address: Address{City: "Dhaka"}}
emp.Name    // promoted — same as emp.User.Name
emp.City    // promoted — same as emp.Address.City
emp.Greet() // User's methods also promoted!

// ── Anonymous struct — great for test cases
tests := []struct{ input string; want int }{
    {"42", 42},
    {"-1", -1},
}

// ── Methods
func (u User) Greet() string {      // value receiver — read only
    return "Hi, " + u.Name
}
func (u *User) SetName(n string) {  // pointer receiver — modifies struct
    u.Name = n
}
// Rule: if ANY method uses pointer receiver, use pointer for ALL methods

// ── Constructor pattern
func NewUser(name, email string) (*User, error) {
    if name == "" { return nil, errors.New("name required") }
    return &User{Name: name, Email: email, CreatedAt: time.Now()}, nil
}

PHASE 04 — Functions Deep Dive

Timeline: Week 2

// ── Multiple return values (idiomatic Go)
func divide(a, b float64) (float64, error) {
    if b == 0 { return 0, errors.New("divide by zero") }
    return a / b, nil
}
result, err := divide(10, 2)
if err != nil { log.Fatal(err) }

// ── Named return values
func minMax(nums []int) (min, max int) {
    min, max = nums[0], nums[0]
    for _, n := range nums[1:] {
        if n < min { min = n }
        if n > max { max = n }
    }
    return // naked return
}

// ── Variadic
func sum(nums ...int) int {   // nums is []int inside
    total := 0
    for _, n := range nums { total += n }
    return total
}
sum(1, 2, 3)
sum(mySlice...)   // spread existing slice

// ── First-class functions
fn := func(n int) int { return n * 2 }

// Higher-order — function as parameter
func apply(nums []int, fn func(int) int) []int {
    res := make([]int, len(nums))
    for i, v := range nums { res[i] = fn(v) }
    return res
}
doubled := apply([]int{1,2,3}, func(n int) int { return n * 2 })

// ── Closures — capture variables from outer scope
func counter() func() int {
    count := 0
    return func() int {
        count++
        return count
    }
}
c := counter()
c() // 1
c() // 2

// ── Functional Options Pattern (used in most Go libraries)
type Server struct { host string; port int; timeout time.Duration }

func WithPort(p int) func(*Server) {
    return func(s *Server) { s.port = p }
}

func NewServer(host string, opts ...func(*Server)) *Server {
    s := &Server{host: host, port: 8080, timeout: 30*time.Second}
    for _, opt := range opts { opt(s) }
    return s
}
srv := NewServer("localhost", WithPort(9090))

// ── panic / recover
// panic: for programmer errors (nil dereference, impossible state)
// NOT for regular errors — use error returns instead
func mustParse(s string) int {
    n, err := strconv.Atoi(s)
    if err != nil { panic(fmt.Sprintf("mustParse: %v", err)) }
    return n
}

// recover: must be in a deferred function
func safeDiv(a, b int) (result int, err error) {
    defer func() {
        if r := recover(); r != nil {
            err = fmt.Errorf("recovered: %v", r)
        }
    }()
    return a / b, nil  // panics if b==0, recover catches it
}

PHASE 05 — OOP in Go: Interfaces & Pointers

Timeline: Week 3

Interfaces — Implicit Satisfaction 

// Define behavior (not type)
type Shape interface {
    Area() float64
    Perimeter() float64
}

// Circle satisfies Shape — automatically, no "implements" keyword!
type Circle struct { Radius float64 }
func (c Circle) Area() float64      { return math.Pi * c.Radius * c.Radius }
func (c Circle) Perimeter() float64 { return 2 * math.Pi * c.Radius }

type Rectangle struct { Width, Height float64 }
func (r Rectangle) Area() float64      { return r.Width * r.Height }
func (r Rectangle) Perimeter() float64 { return 2 * (r.Width + r.Height) }

// Polymorphism
func printInfo(s Shape) {
    fmt.Printf("Area: %.2f, Perimeter: %.2f\n", s.Area(), s.Perimeter())
}
shapes := []Shape{Circle{5}, Rectangle{4, 6}}
for _, s := range shapes { printInfo(s) }

// ── Composing interfaces
type ReadWriter interface {
    io.Reader  // embed: Read(p []byte) (n int, err error)
    io.Writer  // embed: Write(p []byte) (n int, err error)
}

// ── Empty interface (any)
var val any = "hello"   // can hold any value
var val any = 42

// ── Type assertion
var i interface{} = "hello"
s := i.(string)          // panics if wrong type
s, ok := i.(string)      // safe — ok=false if wrong type
n, ok := i.(int)         // ok=false, n=0

// ── Type switch
func describe(i interface{}) {
    switch v := i.(type) {
    case int:    fmt.Printf("int: %d\n", v)
    case string: fmt.Printf("string: %s\n", v)
    case bool:   fmt.Printf("bool: %t\n", v)
    case []int:  fmt.Printf("[]int len=%d\n", len(v))
    default:     fmt.Printf("unknown: %T\n", v)
    }
}

// ── Key stdlib interfaces (implement these on your types)
// error:         Error() string
// io.Reader:     Read(p []byte) (n int, err error)
// io.Writer:     Write(p []byte) (n int, err error)
// fmt.Stringer:  String() string  — custom print via fmt
// sort.Interface: Len() int, Less(i,j int) bool, Swap(i,j int)
// http.Handler:  ServeHTTP(ResponseWriter, *Request)

// Implement fmt.Stringer
func (u User) String() string {
    return fmt.Sprintf("User{%d, %s}", u.ID, u.Name)
}
fmt.Println(u)  // calls String() automatically

💡 Go proverb: "The bigger the interface, the weaker the abstraction." Keep interfaces small — io.Reader has 1 method. error has 1 method.

Pointers — Complete

// Basics
x := 42
p := &x          // p is *int — holds address of x
*p = 100         // dereference — modify x through pointer
fmt.Println(x)   // 100

// new() vs &Type{}
p1 := new(int)          // allocates *int, zero value
u  := &User{Name: "A"}  // preferred over new(User)

// When to use pointer receivers:
// 1. Method modifies the struct
func (u *User) SetName(n string) { u.Name = n }
// 2. Large struct (avoid copying)
func (d *BigData) Process() {}
// 3. Consistency: if any method uses pointer, all should

// Nil pointer — most common panic!
var u *User         // nil
// u.Name = "Alice" // PANIC: nil pointer dereference
if u != nil { u.Name = "Alice" }  // safe

// Pointer to optional value — common pattern
type Config struct {
    Timeout  *int    // nil = use default
    MaxConns *int
}
// Helper function
func Ptr[T any](v T) *T { return &v }
cfg := Config{Timeout: Ptr(30)}

PHASE 06 — Error Handling (The Go Way)

Timeline: Week 3

// ── No try/catch. Errors are return values.
// JavaScript:
try {
    const data = JSON.parse(raw)
} catch(e) { ... }

// Go:
data, err := json.Unmarshal(raw, &v)
if err != nil { return fmt.Errorf("parse: %w", err) }

// ── Sentinel errors (predefined, like constants)
var (
    ErrNotFound     = errors.New("not found")
    ErrUnauthorized = errors.New("unauthorized")
    ErrConflict     = errors.New("conflict")
)
// Usage
return nil, ErrNotFound
// Check
if errors.Is(err, ErrNotFound) { ... }

// ── Custom error types (with data)
type ValidationError struct {
    Field   string
    Message string
}
func (e *ValidationError) Error() string {
    return fmt.Sprintf("validation: %s — %s", e.Field, e.Message)
}

// ── Error wrapping (%w — Go 1.13+)
func getUser(id int) (*User, error) {
    u, err := dbQuery(id)
    if err != nil {
        return nil, fmt.Errorf("getUser(%d): %w", id, err)  // %w wraps the error
    }
    return u, nil
}

// errors.Is — checks entire chain
if errors.Is(err, ErrNotFound) { ... }
if errors.Is(err, sql.ErrNoRows) { ... }

// errors.As — check type in chain
var ve *ValidationError

// Channel direction in signatures

func consumer(ch <-chan int) { v := <-ch }  // receive-only

// ── select — like switch but for channels

case v := <-ch1:
    process(v)
case ch2 <- x:

    return ctx.Err()
default:
    doOtherWork()   // non-blocking — runs if no case ready
}

for i := 0; i < 10; i++ {

// ── sync.Mutex — exclusive lock
type SafeCounter struct {
    mu    sync.Mutex
    count int
}
func (c *SafeCounter) Inc() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.count++
}

// ── sync.RWMutex — multiple readers OR one writer
type Cache struct {
    mu   sync.RWMutex
    data map[string]string
}
func (c *Cache) Get(key string) string {
    c.mu.RLock()
    defer c.mu.RUnlock()   // multiple readers allowed simultaneously
    return c.data[key]
}
func (c *Cache) Set(key, val string) {
    c.mu.Lock()
    defer c.mu.Unlock()    // exclusive write lock
    c.data[key] = val
}

// ── sync.Once — run exactly once (singleton pattern)
var (
    once     sync.Once
    instance *Database
)
func GetDB() *Database {
    once.Do(func() { instance = newDatabase() })
    return instance
}

// ── sync.Pool — reuse objects, reduce GC pressure
var bufPool = sync.Pool{
    New: func() interface{} { return new(bytes.Buffer) },
}

buf.Reset()
defer bufPool.Put(buf)

// ── errgroup (golang.org/x/sync) — WaitGroup + error propagation

    url := url
    g.Go(func() error {
        return fetch(ctx, url)
    })
}

// ── Semaphore — limit concurrency
sem := make(chan struct{}, 10)   // max 10 concurrent
for _, url := range urls {
    sem <- struct{}{}            // acquire
    go func(u string) {
        defer func() { <-sem }()  // release
        fetch(u)
    }(url)
}

// ── Worker Pool Pattern
func startWorkerPool(jobs <-chan Job, results chan<- Result, n int) {
    var wg sync.WaitGroup
    for i := 0; i < n; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for job := range jobs {
                results <- process(job)
            }
        }()
    }
    go func() { wg.Wait(); close(results) }()
}

func Map[T, U any](slice []T, fn func(T) U) []U {
    result := make([]U, len(slice))
    for i, v := range slice { result[i] = fn(v) }
    return result
}
func Filter[T any](slice []T, pred func(T) bool) []T {
    var result []T
    for _, v := range slice {
        if pred(v) { result = append(result, v) }
    }
    return result
}
func Reduce[T, U any](slice []T, init U, fn func(U, T) U) U {
    acc := init
    for _, v := range slice { acc = fn(acc, v) }
    return acc
}

}

// 1. Define interface in domain layer
type UserRepository interface {
    FindByID(ctx context.Context, id int) (*User, error)
    FindByEmail(ctx context.Context, email string) (*User, error)
    Create(ctx context.Context, u *User) error
    Update(ctx context.Context, u *User) error

// 2. Implement in infrastructure layer
type postgresUserRepo struct{ db *sql.DB }
func (r *postgresUserRepo) FindByID(ctx context.Context, id int) (*User, error) {
    var u User
    err := r.db.QueryRowContext(ctx, "SELECT id, name FROM users WHERE id = $1", id).
        Scan(&u.ID, &u.Name)
    if err == sql.ErrNoRows { return nil, ErrNotFound }
    return &u, err
}

// 3. Service depends on INTERFACE, not implementation
type UserService struct {
    repo UserRepository  // can swap with mock in tests!
    log  *slog.Logger
}
func NewUserService(repo UserRepository, log *slog.Logger) *UserService {
    return &UserService{repo: repo, log: log}
}

type UserHandler struct { svc *UserService }

    if err != nil { c.JSON(500, gin.H{"error": "internal error"}); return }

}

// ── Observer (event system)
type EventBus struct {
    mu          sync.RWMutex
    subscribers map[string][]func(Event)
}
func (b *EventBus) Subscribe(event string, fn func(Event)) {
    b.mu.Lock(); defer b.mu.Unlock()
    b.subscribers[event] = append(b.subscribers[event], fn)
}
func (b *EventBus) Publish(event string, e Event) {
    b.mu.RLock(); defer b.mu.RUnlock()
    for _, fn := range b.subscribers[event] { go fn(e) }
}

// ── Middleware chain (used in HTTP, gRPC, etc.)
type Middleware func(Handler) Handler
func Chain(h Handler, middleware ...Middleware) Handler {
    for i := len(middleware) - 1; i >= 0; i-- {
        h = middleware[i](h)
    }
    return h
}
final := Chain(myHandler, Logger(), Auth(), RateLimit())

type QueryBuilder struct {
    table      string
    conditions []string
    args       []interface{}
    limit      int
}
func (q *QueryBuilder) Where(cond string, args ...interface{}) *QueryBuilder {
    q.conditions = append(q.conditions, cond)
    q.args = append(q.args, args...)
    return q
}
func (q *QueryBuilder) Limit(n int) *QueryBuilder {
    q.limit = n; return q
}

// ── JWT Authentication (go get github.com/golang-jwt/jwt/v5)
type Claims struct {
    UserID int    `json:"user_id"`
    Email  string `json:"email"`
    jwt.RegisteredClaims
}

func GenerateToken(userID int, email string) (string, error) {
    claims := Claims{
        UserID: userID,
        Email:  email,
        RegisteredClaims: jwt.RegisteredClaims{
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(24 * time.Hour)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
        },
    }
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString([]byte(os.Getenv("JWT_SECRET")))
}

func ValidateToken(tokenString string) (*Claims, error) {
    token, err := jwt.ParseWithClaims(tokenString, &Claims{}, func(t *jwt.Token) (interface{}, error) {
        if _, ok := t.Method.(*jwt.SigningMethodHMAC); !ok {
            return nil, fmt.Errorf("unexpected signing method: %v", t.Header["alg"])
        }
        return []byte(os.Getenv("JWT_SECRET")), nil
    })
    if err != nil { return nil, err }
    return token.Claims.(*Claims), nil
}
